SCSS 
$corPadrao: #cc6699;
$tamanhoDaFontePadrao: 14px;

@mixin tipografia($cor: $corPadrao){
	font-family: 'Raleway', sans-serif;
	font-weight: 400;
	color: $cor;
}

$listaDeTitulos: 'h6', 'h5','h4', 'h3', 'h2','h1';
@each $titulo in $listaDeTitulos{
	
	$tamanhoDaFontePadrao: $tamanhoDaFontePadrao + 5;
	#{$titulo}{
		@include tipografia();
		text-transform: uppercase;
		text-align: center;
		font-size: $tamanhoDaFontePadrao;
	}
}
